1.Model¶

[3]:

# Create a causal model from the data and given common causes.
xs = ""
for i in range(1,26):
    xs += ("x"+str(i)+"+")

model=CausalModel(
        data = data,
        treatment='treatment',
        outcome='y_factual',
        common_causes=xs.split('+')
        )

)

print("Causal Estimate is " + str(estimate.value))
data_1 = data[data["treatment"]==1]
